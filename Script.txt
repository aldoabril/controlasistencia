
CREATE TABLE tarjeta (
                id INT NOT NULL,
                password VARCHAR NOT NULL,
                PRIMARY KEY (id)
);


CREATE TABLE agencia (
                id INT NOT NULL,
                direccion VARCHAR NOT NULL,
                PRIMARY KEY (id)
);


CREATE TABLE marcador (
                id INT NOT NULL,
                agencia INT NOT NULL,
                nombre VARCHAR NOT NULL,
                ip VARCHAR NOT NULL,
                PRIMARY KEY (id)
);


CREATE TABLE cargo (
                id INT NOT NULL,
                nombre VARCHAR NOT NULL,
                PRIMARY KEY (id)
);


CREATE TABLE empleado (
                id INT NOT NULL,
                cargo INT NOT NULL,
                nombre1 VARCHAR NOT NULL,
                nombre2 VARCHAR NOT NULL,
                apellidop VARCHAR NOT NULL,
                apellidom VARCHAR NOT NULL,
                dni VARCHAR NOT NULL,
                agencia INT NOT NULL,
                tarjeta INT NOT NULL,
                PRIMARY KEY (id)
);


CREATE TABLE rol (
                id INT NOT NULL,
                nombre VARCHAR NOT NULL,
                PRIMARY KEY (id)
);


CREATE TABLE usuario (
                id INT NOT NULL,
                login VARCHAR NOT NULL,
                password VARCHAR NOT NULL,
                rol INT NOT NULL,
                PRIMARY KEY (id)
);


CREATE TABLE horario (
                id INT NOT NULL,
                nombre VARCHAR NOT NULL,
                refrigerio BOOLEAN NOT NULL,
                usuario INT NOT NULL,
                PRIMARY KEY (id)
);


CREATE TABLE dia (
                id INT NOT NULL,
                nombre VARCHAR NOT NULL,
                hora_en TIME NOT NULL,
                hora_sa TIME NOT NULL,
                hora_ren TIME NOT NULL,
                hora_rsa TIME NOT NULL,
                horario INT NOT NULL,
                fecha DATE NOT NULL,
                hora_en_real TIME NOT NULL,
                hora_sa_real TIME NOT NULL,
                hora_ren_real TIME NOT NULL,
                hora_rsa_real TIME NOT NULL,
                PRIMARY KEY (id)
);
